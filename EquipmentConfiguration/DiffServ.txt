# SCIENTIFIC_SWL3

conf t
access-list 181 permit udp 10.90.128.0 0.0.127.255 any
access-list 193 permit udp 10.106.128.0 0.0.127.255 any

class-map match-all EF
class-map match-all 193
class-map match-all AF11
class-map match-all 181

end
write

conf t
policy-map IN_SCI
class EF
set ip dscp 46
class AF11
set ip dscp 10
class AF23
set ip dscp 22

end
write

conf t
int VLAN 181
service-policy input IN_SCI
int VLAN 213
service-policy input IN_SCI

end
write

conf t
class-map match-all PREMIUM
match ip dscp 46
class-map match-all GOLD
match ip dscp 10
class-map match-all SILVER
match ip dscp 22
class-map match-all BEST-EFFORT
match ip dscp 0

end
write

conf t
policy-map OUT_SCI
class PREMIUM
priority percent 40
class GOLD
bandwidth percent 20
class SILVER
bandwidth percent 15
class BEST-EFFORT
police 16000 2000 2000 conform-action set-dscp-transmit 0

end
write

conf t
int VLAN 224
service-policy output OUT_SCI

end
write
