#     10.64.0.0				0.63.255.255

# AVEIRO_ISP1

primary 10.0.0.1
peer 10.0.0.9
mapping-id 101
end

ip nat inside source list 2 pool AVEIRO_POOL mapping-id 101 overload

int f1/1
ip nat inside
int f0/0
ip nat outside
